Banana=(2*(65.5+2.25))
Strawberry=(Pear*1)
Cherry=(Lemon/2)
Apple=(3*5)
Blueberry=(Pear+0)
Durian=(Apple*Pear)
Lemon=(Pear+Tangerine)
Pear=(Apple+1)
Coconut=(1+0)
Starfruit=(2**4)
Mango=((Apple+(Durian+(Pear*(Blueberry*(Coconut/Strawberry)))))/2)
Grape=(Banana+Lemon)
Raspberry=(Cherry-1)
Pineapple=(Mango+Strawberry)
Watermelon=(Banana*Cherry)
Papaya=(Lemon+Pear)
Kiwi=(Apple/Coconut)
Orange=(Blueberry+Durian)
Peach=(Starfruit-Grape)
Pomegranate=(Banana*Raspberry)
Apricot=(Pineapple/Orange)
Fig=(Mango+Papaya)
Guava=(Kiwi*Coconut)
Avocado=(Pear+Pomegranate)
Cranberry=(Lemon-Apricot)
Blackberry=(Starfruit*Grape)
Dragonfruit=((Cherry*3)+(Coconut/2))
Lychee=((Mango-5)*(Banana+Apple))
Passionfruit=((Lemon+Pineapple)/(Kiwi+1))
Mulberry=((Durian+Raspberry)-(Papaya*2))
Clementine=((Blackberry/Grape)+(Guava-Fig))
Acai=((Pomegranate*2)-(Cranberry/3))
Gooseberry=((Starfruit**3)+(Watermelon-Peach))
Rambutan=((Avocado/Apricot)*(Strawberry+Blueberry))
Kumquat=((Lychee+Passionfruit)/(Mulberry-Clementine))
Tamarind=((Acai+Gooseberry)*(Rambutan/Kumquat))
